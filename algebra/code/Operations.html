293     private void jToggleButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                               
294 
295         double values1[][] = new double[x1][y1];
296         double values2[][] = new double[x2][y2];
297 
298         for (int i = 0; i < x1; i++) {
299             for (int j = 0; j < y1; j++) {
300                 if (table1.getValueAt(i, j) == null) {
301                     System.out.println("Value = null");
302                     return;
303                 }
304                 values1[i][j] = Double.parseDouble("" + table1.getValueAt(i, j));
305             }
306         }
307 
308         for (int i = 0; i < x2; i++) {
309             for (int j = 0; j < y2; j++) {
310                 if (table2.getValueAt(i, j) == null) {
311                     System.out.println("Value = null");
312                     return;
313                 }
314                 values2[i][j] = Double.parseDouble("" + table2.getValueAt(i, j));
315             }
316         }
317         double values3[][];
318         if (plu.isSelected()) {
319             if (x1 != x2 || y1 != y2) {
320                 err.setText("x1 != x2 or y1 != y2");
321                 return;
322             } else {
323                 err.setText("");
324                 values3 = new double[x1][y1];
325                 for (int i = 0; i < x1; i++) {
326                     for (int j = 0; j < y1; j++) {
327                         values3[i][j] = values1[i][j] + values2[i][j];
328                     }
329                 }
330             }
331 
332             Object[][] objects = new Object[x1][y1];
333             String titles[] = new String[y1];
334             for (int i = 0; i < y1; i++) {
335                 titles[i] = ""+(i+1);
336             }
337             for (int i = 0; i < x1; i++) {
338                 for (int j = 0; j < y1; j++) {
339                     objects[i][j] = (Object) values3[i][j];
340                 }
341             }
342             table3.setModel(new javax.swing.table.DefaultTableModel(objects, titles));
343         } else if (min.isSelected()) {
344             if (x1 != x2 || y1 != y2) {
345                 err.setText("x1 != x2 or y1 != y2");
346                 return;
347             } else {
348                 err.setText("");
349                 values3 = new double[x1][y1];
350                 for (int i = 0; i < x1; i++) {
351                     for (int j = 0; j < y1; j++) {
352                         values3[i][j] = values1[i][j] - values2[i][j];
353                     }
354                 }
355             }
356 
357             Object[][] objects = new Object[x1][y1];
358             String titles[] = new String[y1];
359             for (int i = 0; i < y1; i++) {
360                 titles[i] = ""+(i+1);
361             }
362             for (int i = 0; i < x1; i++) {
363                 for (int j = 0; j < y1; j++) {
364                     objects[i][j] = (Object) values3[i][j];
365                 }
366             }
367             table3.setModel(new javax.swing.table.DefaultTableModel(objects, titles));
368         } 
369         else if (mul.isSelected()) {
370             if (y1 != x2) {
371                 err.setText("y1 does not equal x2");
372                 return;
373             } else {
374                 Object[][] objects = new Object[x2][y1];
375                 String titles[] = new String[y1];
376                 for(int i = 0; i < y1; i++){
377                     titles[i] = ""+(i+1);                
378                 }
379                 values3 = Algebra.multiply(values2, values1);
380                 for (int i = 0; i < x2; i++) {
381                     for (int j = 0; j < y1; j++) {
382                         objects[i][j] = (Object) values3[i][j];
383                     }
384                 }
385                 table3.setModel(new javax.swing.table.DefaultTableModel(objects, titles));
386             }
387         }
