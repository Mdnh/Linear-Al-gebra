380     private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {                                         
381         
382         x = Integer.parseInt(X.getText());
383         y = Integer.parseInt(Y.getText());
384         save();
385         titles = new String [x]; for(int i = 0; i < x; i++) titles[i] = "x" + (i+1);
386         double inverted [][] = new double [y][x];
387         for(int i = 0; i < x; i++){
388             for(int j = 0; j < y; j++){
389                 inverted [j][i] = values [i][j];
390             }
391         }
392         values = inverted.clone();
393         Objects = new Object [y][x];
394         for(int i = 0; i < values.length; i++){
395             for(int j = 0; j < values[0].length; j++){
396                 Objects [i][j] = (Object)values[i][j];
397             }
398         }
399         X.setText(""+y);
400         Y.setText(""+x);
401         table1.setModel(new javax.swing.table.DefaultTableModel(Objects, titles));
402     }
